} function normalizeTeamName(name) { return String(name || "") .replace(/\([^)]*\)/g, "") .replace(/[:*]/g, "") .replace(/[^a-zA-Z0-9 ]/g, " ") .replace(/\s+/g, " ") .trim() .replace(/^bullets$/i, "storm") .toLowerCase();
